        // Thread interaction functionality
        document.addEventListener('DOMContentLoaded', function() {
            // Message hover effects
            const messages = document.querySelectorAll('.message-hover');
            messages.forEach(message => {
                message.addEventListener('mouseenter', function() {
                    this.style.transform = 'translateX(4px)';
                    this.style.boxShadow = '0 4px 12px rgba(0, 0, 0, 0.1)';
                });
                
                message.addEventListener('mouseleave', function() {
                    this.style.transform = 'translateX(0)';
                    this.style.boxShadow = '';
                });
            });

            // Thread navigation clicks
            const navItems = document.querySelectorAll('#sidebar .cursor-pointer');
            navItems.forEach(item => {
                item.addEventListener('click', function() {
                    // Remove active state from all items
                    navItems.forEach(nav => nav.classList.remove('bg-blue-100', 'border-blue-200'));
                    // Add active state to clicked item
                    this.classList.add('bg-blue-100', 'border-blue-200');
                });
            });

            // Expand/Collapse functionality
            const expandBtn = document.querySelector('button:has(i.fa-expand)');
            const collapseBtn = document.querySelector('button:has(i.fa-compress)');
            
            if (expandBtn) {
                expandBtn.addEventListener('click', function() {
                    const messages = document.querySelectorAll('#thread-content .prose');
                    messages.forEach(msg => {
                        msg.style.maxHeight = 'none';
                        msg.style.overflow = 'visible';
                    });
                });
            }

            if (collapseBtn) {
                collapseBtn.addEventListener('click', function() {
                    const messages = document.querySelectorAll('#thread-content .prose');
                    messages.forEach(msg => {
                        msg.style.maxHeight = '120px';
                        msg.style.overflow = 'hidden';
                    });
                });
            }
        });
